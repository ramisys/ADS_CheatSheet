2. Functions

String Functions

SELECT CONCAT('Hello', ' ', 'World');
SELECT UPPER('mysql');
SELECT LOWER('MYSQL');

Numeric Functions

SELECT ABS(-25);
SELECT CEIL(4.2);
SELECT FLOOR(4.8);

Date Functions

SELECT NOW();
SELECT CURDATE();
SELECT DATE_ADD(CURDATE(), INTERVAL 7 DAY);

IFNULL & COALESCE

SELECT IFNULL(middle_name, 'N/A') FROM users;
SELECT COALESCE(NULL, NULL, 'First non-null');

IF Function

SELECT name, IF(salary > 5000, 'High', 'Low') AS salary_status
FROM employees;

CASE Operator

SELECT name,
       CASE 
         WHEN salary > 5000 THEN 'High'
         WHEN salary BETWEEN 3000 AND 5000 THEN 'Medium'
         ELSE 'Low'
       END AS salary_status
FROM employees;
